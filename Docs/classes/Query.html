Defined in src\wmi-query.js:43

Parameters:

  • options Object

    The object whose properties will be used to init the query

    • [timeout=5000] Number optional

      time out for query execution

    • [node] String | Array optional

      node(s) to query, can contains multiples server name separated by comma: "pc1,pc2" or array of node

    • [verb] String optional

      verb of the query, must be in Query.SUPPORTED_VERBS

    • [alias] String optional

      alias being queried

    • [where] String | Array optional

      where clause, query will add parenthesis around it

    • [field] String | Array optional

      field being queried, used mainly by "get" verb

    • [action] String optional

      action being executed, use mainly by "call" verb

    • [format='JSON'] String optional

      query output, must be in Query.SUPPORTED_FORMATS or be a valid path to a XSL file (in this case the output will default to raw).

    • [cmd] String optional

      manual command, will override mosts others options

    • [parser] String | Function optional

      Function that will be used to parse cmd output

    • [help=false] Boolean optional

      flag used when the cmd executed needs the "/?" option

Example:

//requiring module
var wq = require("wmi-query");
//implements of query describe in first link
//using static get method
wq.get({"alias":"baseboard", "field":"Manufacturer, Model, Name, PartNumber, slotlayout, serialnumber, poweredon"}, function(r) {
    //...
});
//manual execution and field as array
var myQuery = new wq({"alias":"bios", "verb":"get", "field": ["name", "version", "serialnumber"]})
myQuery.exec(function(r) { 
    // ... 
});
